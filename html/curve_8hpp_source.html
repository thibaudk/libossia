251 template <typename X, typename Y>
252 inline Y curve<X, Y>::get_y0() const
253 {
254  if (!m_y0_destination)
255  return m_y0;
256  else
257  {
258  if (m_y0_cacheUsed)
259  return m_y0_cache;
260 
261  const destination& dest = *m_y0_destination;
262  m_y0_cacheUsed = true;
263  m_y0_cache = convert_to_template_type_value(
264  dest.address().fetch_value(), dest.index.begin());
265 
266  if (m_scaleBounds)
267  {
268  scale_info& bounds = *m_scaleBounds;
269  bounds.start = m_y0_cache;
270 
271  // Rescale the whole curve with the acquired information.
272  // It is currently in [0; 1].
273  if (bounds.start < bounds.min)
274  bounds.min = bounds.start;
275  else if (bounds.start > bounds.max)
276  bounds.max = bounds.start;
277 
278  if (bounds.start < bounds.end)
279  {
280  for (auto& pt : m_points.container)
281  {
282  Y& y = pt.second.first;
283  y = ossia::easing::ease{}(bounds.min, bounds.max, y);
284  }
285  }
286  else if (bounds.start == bounds.end)
287  {
288  m_points.clear();
289  m_x0 = 0;
290  }
291  else // start > end
292  {
293  for (auto& pt : m_points.container)
294  {
295  auto& y = pt.second.first;
296  const auto fun = [=](Y val) -> Y {
297  return bounds.max - val * (bounds.max - bounds.min);
298  };
299  y = fun(y);
300  }
301  }
302  }
303  return m_y0_cache;
304  }
